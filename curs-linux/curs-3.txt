2.
# listare fisiere / directoare

3. Metacaractere:
*   - 0 sau mai multe caractere
?   - 1 caracter
^   - negare

ls a[1-9].jpg
ls a[137].jpg

4. Redirectionari / Piping
comanda > fisier # stdout
comanda 2> fisier # stderr
comanda >> fisier # stdout append

5. Piping
com1 | com2 | com3
ping -c 2 c.ro | tail -n 2 | head -n 1| cut -d"," -f3 | cut -d"%" -f1

rm

-r  # recursiv
-i  # interactiv
-f  # force
